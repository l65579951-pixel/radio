// Referencias a los elementos del DOM (Asegúrate de que coincidan con tus IDs de HTML) const playBtn = document.getElementById('btn-play'); const stopBtn = document.getElementById('btn-stop'); const powerLed = document.getElementById('power-led'); const leftReel = document.getElementById('left-reel'); const rightReel = document.getElementById('right-reel'); const trackName = document.getElementById('track-name'); const urlInput = document.getElementById('spotify-url'); let spotifyController = null; let isPlaying = false; const defaultPlaylist = "https://open.spotify.com/playlist/37i9dQZF1DX8Uebhn9vU6P"; window.onload = () => { // Cargar tema guardado const savedTheme = localStorage.getItem('theme') || 'light'; document.documentElement.setAttribute('data-theme', savedTheme); if (urlInput) urlInput.value = defaultPlaylist; }; // 1. Inicialización ultra-segura de la API de Spotify window.onSpotifyIframeApiReady = (IFrameAPI) => { const element = document.getElementById('spotify-embed-container'); if (!element) { console.error("Error: No se encontró el contenedor '#spotify-embed-container' en tu HTML."); return; } const options = { width: '100%', height: '80', // Altura del widget compacto de Spotify uri: getSpotifyUri(defaultPlaylist) }; IFrameAPI.createController(element, options, (EmbedController) => { spotifyController = EmbedController; if (trackName) trackName.textContent = "Cinta lista. ¡Dale a PLAY!"; // Escucha de eventos nativos de Spotify spotifyController.on('playback_update', e => { if (!e || !e.data) return; const duration = e.data.duration || 0; const position = e.data.position || 0; isPlaying = !e.data.isPaused; setVisualState(isPlaying); if (trackName) { if (isPlaying) { trackName.textContent = duration > 0 ? `Sonando... (${formatTime(position)} / ${formatTime(duration)})` : "Reproduciendo..."; } else { trackName.textContent = "Cinta pausada"; } } }); }); }; // 2. Extractor de URIs a prueba de errores function getSpotifyUri(url) { if (!url) return `spotify:playlist:37i9dQZF1DX8Uebhn9vU6P`; try { const match = url.match(/(track|playlist|album|artist)\/([a-zA-Z0-9]+)/); if (match && match[1] && match[2]) { return `spotify:${match[1]}:${match[2]}`; } } catch (e) { console.warn("Error parseando la URL, usando playlist por defecto.", e); } return `spotify:playlist:37i9dQZF1DX8Uebhn9vU6P`; } // 3. Cargar nueva cinta en caliente function loadSpotifyUri() { if (!spotifyController) { alert("Espera a que el reproductor cargue por completo."); return; } const url = urlInput ? urlInput.value.trim() : ""; const uri = getSpotifyUri(url); spotifyController.loadUri(uri); setVisualState(false); if (trackName) trackName.textContent = "Cargando nueva cinta..."; } // 4. Controles físicos (con condicionales para evitar que rompan la página) function pressPlay() { if (!spotifyController) { console.warn("El controlador de Spotify no está listo."); return; } // Usamos métodos explícitos para esquivar incompatibilidades del viejo togglePlay() if (isPlaying) { spotifyController.pause(); } else { spotifyController.play(); } } function pressStop() { if (!spotifyController) return; spotifyController.pause(); setVisualState(false); // Animación mecánica del botón STOP if (stopBtn) { stopBtn.classList.add('active'); setTimeout(() => stopBtn.classList.remove('active'), 250); } } // 5. Renderizador de estados visuales (CSS classes) function setVisualState(playing) { if (playing) { if (playBtn) playBtn.classList.add('active', 'active-play'); if (stopBtn) stopBtn.classList.remove('active'); if (powerLed) powerLed.classList.add('active'); if (leftReel) leftReel.classList.add('spinning'); if (rightReel) rightReel.classList.add('spinning'); } else { if (playBtn) playBtn.classList.remove('active', 'active-play'); if (powerLed) powerLed.classList.remove('active'); if (leftReel) leftReel.classList.remove('spinning'); if (rightReel) rightReel.classList.remove('spinning'); } } // Formateador de minutos y segundos function formatTime(ms) { if (isNaN(ms) || ms < 0) return "0:00"; const minutes = Math.floor(ms / 60000); const seconds = Math.floor((ms % 60000) / 1000); return `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`; }
