// 1. Referencias al DOM (Validadas) const playBtn = document.getElementById('btn-play'); const stopBtn = document.getElementById('btn-stop'); const powerLed = document.getElementById('power-led'); const leftReel = document.getElementById('left-reel'); const rightReel = document.getElementById('right-reel'); const trackName = document.getElementById('track-name'); const urlInput = document.getElementById('spotify-url'); let spotifyController = null; let isPlaying = false; const defaultPlaylist = "https://open.spotify.com/playlist/37i9dQZF1DX8Uebhn9vU6P"; // 2. Definir la función de inicialización ANTES de cargar el script de Spotify window.onSpotifyIframeApiReady = (IFrameAPI) => { const element = document.getElementById('spotify-embed-container'); if (!element) return; const options = { width: '100%', height: '80', uri: getSpotifyUri(defaultPlaylist) }; IFrameAPI.createController(element, options, (EmbedController) => { spotifyController = EmbedController; if (trackName) trackName.textContent = "Cinta cargada. ¡Dale a PLAY!"; // Escuchar actualizaciones de Spotify spotifyController.on('playback_update', e => { if (!e || !e.data) return; const duration = e.data.duration || 0; const position = e.data.position || 0; isPlaying = !e.data.isPaused; setVisualState(isPlaying); if (trackName) { if (isPlaying) { trackName.textContent = duration > 0 ? `Sonando... (${formatTime(position)} / ${formatTime(duration)})` : "Reproduciendo..."; } else { trackName.textContent = "Pausado"; } } }); }); }; // 3. CARGA DINÁMICA DEL SCRIPT (Esto soluciona el problema de sincronización) const script = document.createElement('script'); script.src = "https://open.spotify.com/embed/iframe-api"; script.async = true; document.head.appendChild(script); // 4. Inicializar Tema y URL al cargar la página window.addEventListener('DOMContentLoaded', () => { const savedTheme = localStorage.getItem('theme') || 'light'; document.documentElement.setAttribute('data-theme', savedTheme); if (urlInput) urlInput.value = defaultPlaylist; }); // 5. Extractor de URIs function getSpotifyUri(url) { if (!url) return `spotify:playlist:37i9dQZF1DX8Uebhn9vU6P`; try { const match = url.match(/(track|playlist|album|artist)\/([a-zA-Z0-9]+)/); if (match && match[1] && match[2]) { return `spotify:${match[1]}:${match[2]}`; } } catch (e) { console.warn("URL de Spotify no soportada, usando default."); } return `spotify:playlist:37i9dQZF1DX8Uebhn9vU6P`; } // 6. Cargar nueva cinta function loadSpotifyUri() { if (!spotifyController) return; const url = urlInput ? urlInput.value.trim() : ""; const uri = getSpotifyUri(url); spotifyController.loadUri(uri); setVisualState(false); if (trackName) trackName.textContent = "Cargando cinta..."; } // 7. Botón PLAY (Con bypass para que la casetera gire pase lo que pase) function pressPlay() { if (!spotifyController) { // Si la API aún no cargó, simulamos el movimiento visual de la casetera igual isPlaying = !isPlaying; setVisualState(isPlaying); if (trackName) trackName.textContent = isPlaying ? "Reproduciendo (Simulado)..." : "Pausado"; return; } try { // Intentamos usar la API oficial spotifyController.togglePlay(); } catch (err) { // Si el navegador bloquea el autoplay por falta de interacción en el iframe: console.warn("Spotify bloqueó la reproducción automática. Simulando estado visual..."); isPlaying = !isPlaying; setVisualState(isPlaying); if (trackName) trackName.textContent = isPlaying ? "Reproduciendo..." : "Pausado"; } } // 8. Botón STOP function pressStop() { if (spotifyController) { try { if (isPlaying) { spotifyController.togglePlay(); } } catch (err) { console.warn("Error al intentar pausar vía API."); } } isPlaying = false; setVisualState(false); if (trackName) trackName.textContent = "Cinta detenida"; if (stopBtn) { stopBtn.classList.add('active'); setTimeout(() => stopBtn.classList.remove('active'), 250); } } // 9. Renderizador visual de la casetera function setVisualState(playing) { if (playing) { if (playBtn) playBtn.classList.add('active', 'active-play'); if (stopBtn) stopBtn.classList.remove('active'); if (powerLed) powerLed.classList.add('active'); if (leftReel) leftReel.classList.add('spinning'); if (rightReel) rightReel.classList.add('spinning'); } else { if (playBtn) playBtn.classList.remove('active', 'active-play'); if (powerLed) powerLed.classList.remove('active'); if (leftReel) leftReel.classList.remove('spinning'); if (rightReel) rightReel.classList.remove('spinning'); } } // Formateador de tiempo function formatTime(ms) { if (isNaN(ms) || ms < 0) return "0:00"; const minutes = Math.floor(ms / 60000); const seconds = Math.floor((ms % 60000) / 1000); return `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`; }
